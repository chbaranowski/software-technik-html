        @Stateless(name="Calculator", mappedName="ejb/Calculator")
        public class CalculatorImpl implements CalculatorRemote, CalculatorLocal {
        
            @PersistenceContext
            private EntityManager em;
            
            private Long id = new Long(1);
            
            public int sum(int add1, int add2) {
                Result result = new Result();
                result.setId(id++);
                result.setAdd1(add1);
                result.setAdd2(add2);
                result.setResult(add1+add2);
                em.persist(result);
                
                Query query = em.createQuery("SELECT e FROM Result e");
                List results = query.getResultList();
                int summe = 0;
                for (Result r : results) {
                    summe += r.getResult();
                }
                return summe;
            }
        
        }
